CREATE TABLE `project_tasks` (
  `id` int(11) NOT NULL AUTO_INCREMENT,
  `siteId` int(11) NOT NULL,
  `title` varchar(255) NOT NULL,
  `description` text DEFAULT NULL,
  `startDate` datetime DEFAULT NULL,
  `endDate` datetime DEFAULT NULL,
  `status` varchar(255) DEFAULT 'Pending',
  `assignedTo` varchar(255) DEFAULT NULL,
  `priority` varchar(255) DEFAULT 'Medium',
  `color` varchar(255) DEFAULT '#3b82f6',
  `progress` float DEFAULT 0,
  `createdAt` datetime NOT NULL,
  `updatedAt` datetime NOT NULL,
  `enteredBy` varchar(255) DEFAULT NULL,
  `editedBy` varchar(255) DEFAULT NULL,
  PRIMARY KEY (`id`),
  KEY `siteId` (`siteId`),
  CONSTRAINT `project_tasks_ibfk_1` FOREIGN KEY (`siteId`) REFERENCES `sites` (`id`) ON DELETE NO ACTION ON UPDATE CASCADE
) ENGINE=InnoDB AUTO_INCREMENT=5 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_general_ci
